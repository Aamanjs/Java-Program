package pr1;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStream;
import java.io.InputStreamReader;
import java.io.OutputStream;
import java.io.PrintStream;
import java.net.ServerSocket;
import java.net.Socket;

public class Server {
	
	public static void main(String[] args) throws IOException {
		
		ServerSocket ss = new ServerSocket(888);
		Socket s = ss.accept();
		System.out.println("got connect from client to server");
		
		//read from socket
		InputStream is = s.getInputStream();
		InputStreamReader ir = new InputStreamReader(is);
		BufferedReader br = new BufferedReader(ir);
		String cmsg = br.readLine();
		
		//write to console 
		System.out.println("message recieved from client is: "+cmsg);
		
		//read from console
		InputStreamReader ir1 = new InputStreamReader(System.in);
		BufferedReader br1 = new BufferedReader(ir1);
		
		System.out.println("Enter the message to Client");
		String smsg = br1.readLine();
		
		//write to socket
		OutputStream os1 = s.getOutputStream();
		PrintStream pr1 = new PrintStream(os1);
		pr1.println(smsg);
		
		
		
	}

}
